git add <filename> - adds specific file to staging
git add . - adds ALL changed files to staging

git commit -m "commit message" - commits all staged files

git push - pushes commits to remote repo

git pull - pulls latest changes of current branch

git branch - displays all availabl branches. '*' indicates your current branch

git checkout - changes connection to desired branch

git branch -d <branch name> - delete a branch

git status - displayed the status of your branch/staged changes/commits

git restore --staged <filename> - removes specific file from staging
